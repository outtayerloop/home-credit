 45def position_target_column(train):

 88# ## Taking care of missing data
 89
 90# + id="m7aKDXPjF-C3"
 91def impute_train_missing_data(train):
 92    """
 93  Return tuple containing train dataframe with median imputed in place of missing numerical values
 94  and a Series with its numerical columns.
 95
 96  Keyword arguments:
 97  train -- the train dataframe
 98  """
 99    imputer = SimpleImputer(missing_values=np.nan, strategy='median')
100    x_dtypes_train = train.dtypes[:-1]
101    num_cols_train = x_dtypes_train == np.number
102    X_train = train.iloc[:, :-1].values
103    imputer.fit(X_train[:, num_cols_train])
104    X_train[:, num_cols_train] = imputer.transform(X_train[:, num_cols_train])
105    train.iloc[:, :-1] = X_train
106    return train, num_cols_train
107

109# + id="bgQxbY5Mjef1"
110def impute_test_missing_data(test):
111    """
112  Return tuple containing test dataframe with median imputed in place of missing numerical values
113  and a Series with its numerical columns.
114
115    Keyword arguments:
116    test -- the test dataframe
117    """
118    imputer = SimpleImputer(missing_values=np.nan, strategy='median')
119    x_dtypes_test = test.dtypes
120    num_cols_test = x_dtypes_test == np.number
121    X_test = test.iloc[:, :].values
122    imputer.fit(X_test[:, num_cols_test])
123    X_test[:, num_cols_test] = imputer.transform(X_test[:, num_cols_test])
124    test.iloc[:, :] = X_test
125    return test, num_cols_test

129# ### Getting text features Na rows percentage
130
131# + id="9LuAtXAyO4sQ"
132def get_train_na_percentages(train):
133    """
134  Return a Series with the percentage of Na values per columns in train dataframe.
135  Must be called just after impute_train_missing_data().
136
137  Keyword arguments:
138  train -- the train dataframe
139  """
140    na_cols_pctg_train = train[train.columns[train.isna().sum() > 0]].isna().sum() / train.shape[0]
141    return na_cols_pctg_train

167def get_textual_column_indexes(train, test):
168    """Return a tuple containing an ndarray with train and test textual column indexes.
169
170  Keyword arguments:
171  train -- the train dataframe
172  test -- the test dataframe
173  """
174    txt_cols_train = train.select_dtypes('object').columns
175    txt_indexes_train = train.columns.get_indexer(txt_cols_train)
176    txt_cols_test = test.select_dtypes('object').columns
177    txt_indexes_test = test.columns.get_indexer(txt_cols_test)
178    return txt_indexes_train, txt_indexes_test

182def label_encode_train(train, txt_indexes_train):
183    """Return the train dataframe with label-encoded textual features.
184
185  Keyword arguments:
186  train -- the train dataframe
187  txt_indexes_train -- ndarray of train textual column indexes
188  """
189    label_encoder_x = LabelEncoder()
190    X_train = train.iloc[:, :-1].values
191    for i in txt_indexes_train:
192        X_train[:, i] = label_encoder_x.fit_transform(X_train[:, i])
193    train.iloc[:, :-1] = X_train
194    return train

214def one_hot_encode_train(train, txt_indexes_train):
215  """Return the train dataframe with one-hot-encoded textual features.
216
217  Keyword arguments:
218  train -- the train dataframe
219  txt_indexes_train -- ndarray of train textual column indexes
220  """
221  train_dummies = pd.get_dummies(train.iloc[:, txt_indexes_train])
222  train.drop(train.select_dtypes('object').columns, axis=1, inplace=True)
223  train = pd.concat([train, train_dummies], axis=1)
224  train = position_target_column(train)
225  return train

228def one_hot_encode_test(test, txt_indexes_test):
229  """Return the test dataframe with label-encoded textual features.
230
231  Keyword arguments:
232  test -- the test dataframe
233  txt_indexes_test -- ndarray of test textual column indexes
234  """
235  test_dummies = pd.get_dummies(test.iloc[:, txt_indexes_test])
236  test.drop(test.select_dtypes('object').columns, axis=1, inplace=True)
237  test = pd.concat([test, test_dummies], axis=1)
238  return test

241def align_encoded_train_test(train, test):
242    """Return encoded train and test dataframes with same columns (except target that is still in train).
243
244      Keyword arguments:
245      test -- the test dataframe
246      train -- the train dataframe
247      """
248    # Align the training and testing data, keep only columns present in both dataframes
249    target_col = train['TARGET']
250    train, test = train.align(test, join='inner', axis=1)
251    train = pd.concat([train, target_col], axis=1)
252    return train, test

259def standardize_train(train, num_cols_train):
260    """Return the train dataframe with standardized numerical features (not the encoded textual dimensions).
261
262  Keyword arguments:
263  train -- the train dataframe
264  """
265    sc = StandardScaler()
266    X_train = train.iloc[:, :-1].values
267    X_train[:, num_cols_train] = sc.fit_transform(X_train[:, num_cols_train])
268    train.iloc[:, :-1] = X_train

273def standardize_test(test, num_cols_test):
274    """Return the test dataframe with standardized numerical features (not the encoded textual dimensions).
275
276  Keyword arguments:
277  test -- the test dataframe
278  """
279    sc = StandardScaler()  # standardization implies values between approximately -3 and 3
280    X_test = test.iloc[:, :].values
281    X_test[:, num_cols_test] = sc.fit_transform(
282        X_test[:, num_cols_test])  # we don't standardize encoded textual dimensions.
283    test.iloc[:, :] = X_test
284    return test

303# ## Executing the preprocessing workflow
304
305# + id="TCTVgKVsCVsR" pycharm={"name": "#%%\n"}
306def build_features():
307    """Build feature and export train and test dataframes to CSVs"""
308    (train, test) = get_dataframes()
309    train = position_target_column(train)
310    (train, test) = drop_id_column(train, test)
311    test = reorder_test_columns(train, test)
312    (train, num_cols_train) = impute_train_missing_data(train)
313    (test, num_cols_test) = impute_test_missing_data(test)
314    na_cols_pctg_train = get_train_na_percentages(train)
315    (train, test) = drop_textual_feat_na_rows(train, test)
316    train = standardize_train(train, num_cols_train)
317    test = standardize_test(test, num_cols_test)
318    (txt_indexes_train, txt_indexes_test) = get_textual_column_indexes(train, test)
319    train = one_hot_encode_train(train, txt_indexes_train)
320    test = one_hot_encode_test(test, txt_indexes_test)
321    train, test = align_encoded_train_test(train, test)
322    export_dataframes_to_csv_files(train, test)
